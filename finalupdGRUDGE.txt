}


/* clicker */


let score = localStorage.getItem("clickScore") || 0
document.getElementById("score").innerText = score


function addScore(){
score++
localStorage.setItem("clickScore",score)
document.getElementById("score").innerText=score
}


/* snake */


const canvas=document.getElementById("snakeGame")
const ctx=canvas.getContext("2d")


let snake=[{x:200,y:200}]
let food={x:100,y:100}
let dx=20
let dy=0
let snakeScore=0


document.addEventListener("keydown",e=>{


if(e.key==="ArrowUp" && dy===0){dx=0;dy=-20}
if(e.key==="ArrowDown" && dy===0){dx=0;dy=20}
if(e.key==="ArrowLeft" && dx===0){dx=-20;dy=0}
if(e.key==="ArrowRight" && dx===0){dx=20;dy=0}


})


function resetSnake(){
snake=[{x:200,y:200}]
dx=20
dy=0
snakeScore=0
document.getElementById("snakeScore").innerText=0
}


function drawSnake(){


if(!document.getElementById("snake").classList.contains("active")) return


ctx.clearRect(0,0,400,400)


snake.unshift({x:snake[0].x+dx,y:snake[0].y+dy})


if(
snake[0].x < 0 ||
snake[0].x >= 400 ||
snake[0].y < 0 ||
snake[0].y >= 400
){
resetSnake()
}


for(let i=1;i<snake.length;i++){
if(snake[i].x===snake[0].x && snake[i].y===snake[0].y){
resetSnake()
}
}


if(snake[0].x===food.x && snake[0].y===food.y){


food.x=Math.floor(Math.random()*20)*20
food.y=Math.floor(Math.random()*20)*20


snakeScore++
document.getElementById("snakeScore").innerText=snakeScore


}else{
snake.pop()
}


ctx.fillStyle="lime"
snake.forEach(p=>ctx.fillRect(p.x,p.y,20,20))


ctx.fillStyle="red"
ctx.fillRect(food.x,food.y,20,20)


}


setInterval(drawSnake,120)


/* pong */


const pong=document.getElementById("pongGame")
const pctx=pong.getContext("2d")


let ballX=200
let ballY=150
let ballDX=3
let ballDY=3
let paddleX=160


pong.addEventListener("mousemove",e=>{
const rect=pong.getBoundingClientRect()
paddleX=e.clientX-rect.left-40
})


function drawPong(){


pctx.clearRect(0,0,400,300)


ballX+=ballDX
ballY+=ballDY


if(ballX<0 || ballX>400) ballDX*=-1
if(ballY<0) ballDY*=-1


if(ballY>280 && ballX>paddleX && ballX<paddleX+80){
ballDY*=-1
}


pctx.fillStyle="white"


pctx.beginPath()
pctx.arc(ballX,ballY,10,0,Math.PI*2)
pctx.fill()


pctx.fillRect(paddleX,280,80,10)


}


setInterval(drawPong,16)
